
class Node:
    def __init__(self,data):
        self.data=data
        self.next=None
        # [40,none]

class Linkedlist:
    def __init__(self):
        self.head=None
        
    def insertfront(self,data):
        newnode = Node(data)
        
        if self.head is None:
            self.head=newnode
            return
        
        newnode.next = self.head
        self.head=newnode
        
    def insertback(self,data):
        newnode = Node(data)
        temp=self.head
        
        while temp.next:
            temp=temp.next
        
        temp.next=newnode
        
    
    def display(self):
        temp = self.head
        while temp:
            print(temp.data,end="->")
            temp=temp.next
        print("None")
            
        
#creating an object for Linkedlist class
items = Linkedlist()
print("insert front")
items.insertfront(10)
items.insertfront(20)
items.insertfront(30)
items.display()
print("insert back")
items.insertback(40)
items.display()
