Analyzing Sentence Complexity in Content Writing

# hello world ai
# hello -> 2v, 3c -> hard
# world -> 1v, 4c -
# ai -> 2v, 0c
# c > v -> hard orelse -> easy
# 8

str = input().split()

if len(str)==0:
  print(0)
else:
  vowels = "aeiou"
  hard=easy=0
  for word in str:
    # hello
    # h e l l o
    v = sum(1 for ch in word if ch in vowels) # 2
    c = sum(1 for ch in word if ch.isalpha() and ch not in vowels) # 3

    if c > v:
      hard = hard + 1
    else:
      easy = easy + 1

  diff = (5 * hard) - (2 * easy)
  print(diff)
