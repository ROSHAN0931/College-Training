Next Leap Year

# Read input
year = int(input())

def is_leap(y):
    return (y % 400 == 0) or (y % 4 == 0 and y % 100 != 0)

# If the given year is leap, return it
if is_leap(year):
    print(year)
else:
    # keep checking next years
    while True:
        year += 1
        if is_leap(year):
            print(year)
            break

-------------------------------------------------------------------------Quality Check for Premium Products

# Read input
N = int(input())

def is_prime(n):
    if n <= 1:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    # check divisors up to sqrt(n)
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

if is_prime(N):
    print("Premium Product")
else:
    print("Regular Product")

    print(1)   # best split: 1+1

    print(2)   # best split: 2+1
